145 public static ArrayList<NetworkLink> getDefaultLinks() {
146 LoggingHelper.LogFunction("Settings Helper : get default links called!");
147 ArrayList<NetworkLink> links = new ArrayList<>();
148 if (hasDefaultLinks()) {
149 for (Object data : ((JSONArray)UNIVERSAL_SETTINGS.get(DEFAULT_LINKS))) {
150 String[] params = data.toString().split("\\|");
151 for (String str : params) {
152 LoggingHelper.LogDebug("Str : "+str);
153 }
154 if (params[params.length-1].equalsIgnoreCase(NetworkLink.LABEL_CSMA)) {
155 links.add(new CSMA(links.size(), params[0], params[1], params[2], params[3], ((params[4].equalsIgnoreCase("Y"))?(true):(false)), true));
156 } else { // default or p2p case...
157 links.add(new P2P(links.size(), params[0], params[1], params[2], params[3], ((params[4].equalsIgnoreCase("Y"))?(true):(false)), true));
158 }
159 }
160 LoggingHelper.LogDebug("Yes! Universal Settings has Default Links!");
161 } else {
162 LoggingHelper.LogDebug("No! Universal Settings has no default links!");
163 }
164
165 LoggingHelper.LogDebug("Returning the links as array list.");
