604 @Override
605 public void keyPressed(KeyEvent e) {

640 // record each mouse click on canvas...
641 // do accordingly...
642 this.painter.addMouseListener(new MouseAdapter() {
643 @Override
644 public void mouseClicked(MouseEvent e) {
645 super.mouseClicked(e);
646 LoggingHelper.LogDebug("Clicked at x : "+e.getX()+" y : "+e.getY());
647 switch (toolStatus) {
648 // if tool is 'node' tool then...
649 case TOOL_NODE : {
650 // create a new node at given point...
651 NodePainter node = new NodePainter(e.getX()-10, e.getY()-10, 20,String.valueOf(painter.getNodes().size()));
652 // add that node to painter for painting on canvas...
653 painter.addAndPrintNode(node);
654 } break;
655
656 // if tool is 'link' tool then...
657 case TOOL_LINK : {
658 int collision = -1;
659 boolean successfulClick = false;
660 // for first successful click....
661 // check if collision with any node...
662 collision = painter.pointCollideWithAny(e.getX(), e.getY());
663 LoggingHelper.LogDebug("Collision : "+collision+" Clicks : "+clicks);
664 if (collision >= 0 && clicks%2!=0) {
665 // if collision then change the information label and increment the no. of clicks...
666 firstNode = collision;
667 lbl_info.setText("Connection Tool Selected: First node selected now, select second node.");
668 successfulClick = true;
669 }
670
671 // for second successful click...
672 // check if collision with any node...
673 if (collision >= 0 && clicks%2==0) {
674 // if collision then change the information label, increment the no. of clicks and open a device configuration dialog box...
675 lbl_info.setText("Connection Tool Selected: To create a link, click on two nodes sequentially.");
676 painter.addAndPrintLink(new P2pLinkPainter(painter.getNodes().get(firstNode), painter.getNodes().get(collision)));
677 successfulClick = true;
678 LoggingHelper.LogDebug(firstNode+" "+collision);
679 instantiateConnectionDialog();
680 ArrayList<Integer> nodes = new ArrayList<>();
681 nodes.add(firstNode);
682 nodes.add(collision);
683 dialogConnection.showDialog(dialogLink.getAllLinks(),dialogNetwork.getAllNetworks(),nodes, toolStatus);
684 firstNode = -1;
685 }
686 incrementClicks(successfulClick);
687 } break;
688
689 case TOOL_LINK_CSMA, TOOL_LINK_WIFI: {
690 int collision = -1;
691
692 collision = painter.pointCollideWithAny(e.getX(), e.getY());
693 LoggingHelper.LogDebug("Collision : "+collision);
694 if (collision >= 0 && !nodesSelected.contains(collision)) {
695 nodesSelected.add(collision);
696 painter.addAndPrintRefNode(new NodePainter(painter.getNodes().get(collision).xPos-5, painter.getNodes().get(collision).yPos-5,30,"",Color.GREEN));
697 }
698 } break;
